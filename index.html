priceplot <- function(data, ticker, normalise=F){

  if (normalise) {
      data  <- data %>% spread(stock, close)

        }

      data <- data %>% gather(key = stock, value =close, -date)
    }

     data %>%
       filter(stock %in% ticker) %>%
       ggplot(aes(x = date, y = close, colour = stock)) +
       geom_line() +
       labs (title = ticker)
}
